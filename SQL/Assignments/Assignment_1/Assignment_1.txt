Q3) Create jobs Table with Constraints
Write an SQL query to create a table named jobs including columns job_id, job_title,
min_salary, max_salary.
Apply constraints:
- job_id must be Primary Key
- job_title cannot be NULL
- max_salary must not exceed 25000
-- create table jobs(
    -> job_id int primary key, job_title varchar(25) not null, min_salary decimal(7,2), max_salary decimal(7,2),
    -> check (max_salary <= 25000));

Q8) Create employees Table with Normal Foreign Key

employee_id, first_name, last_name, email, phone_number, hire_date, job_id,
salary, commission, manager_id and department_id.
-- create table employees(
    -> employee_id int Primary key, first_name varchar(50), last_name varchar(50), email varchar(50), phone_number varchar(10),
    -> hire_date date, job_id int, salary decimal(10,2), commission decimal(10,2), manager_id int, department_id int,
    -> foreign key (department_id) references departments(department_id));
	
Q9) Composite Foreign Key Implementation
	Step 1: Create department_managers Table
	Create a table named department_managers with following structure:
	Field Name Data Type Null Key
	department_id DECIMAL(4,0) NO PRI
	manager_id DECIMAL(6,0) NO PRI
	manager_name VARCHAR(50) NO
	Composite Primary Key: (department_id, manager_id)

	-- create table department_managers(
    -> department_id int not null,
    -> manager_id int unique key not null,
    -> manager_name varchar(50)not null,
    -> Primary key (department_id, manager_id));
		
	Step 2: Apply Composite Foreign Key in employees Table
	Alter the employees table so that the combination (department_id, manager_id)
	must exist in department_managers(department_id, manager_id).
	
	-- alter table employees
    -> add foreign key(manager_id) references department_managers(manager_id);
